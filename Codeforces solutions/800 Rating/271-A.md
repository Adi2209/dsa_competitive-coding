## Question Link :

https://codeforces.com/problemset/problem/271/A

## Code solution :

```cpp

#include<bits/stdc++.h>
using namespace std;
int main(){
    // Input
    int z;
    cin>>z;
    while(true){
        z++;
        // Extracting each digit
        int a=z/1000;
        int b=z/100%10;
        int c=z/10%10;
        int d=z%10;
        // Checking all digits if they are different
        if(a!=b && a!=c && a!=d && b!=c && b!=d && c!=d){
            break;
        }
    }
    cout<<z<<endl;
return 0;
}

```
